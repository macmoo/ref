#include <stdio.h>
int card_conv(unsigned x, int n, char d[])
{
    char dchar[] = "01234567890ABCDEFGHIJKLMNOPQRSTUVWXYX";
    int digits = 0;
    if (x == 0)
        d[digits++] = dchar[0];
    else
    {
        while (x)
        {
            d[digits++] = dchar[x % n];
            x /= n;
        }
    }
    return digits;
}
int main(void)
{
    unsigned no;    // 변환할 정수
    int  cd;        // 변환할 진수
    int  dno;       // 변환후 문자수
    char cno[512];  // 변환후 결과배열
    int  retry;     // 다시?
    puts("10진수를 진수변환하겠다.");
    do
    {
        printf("변환할 정수를 입력하라.:");
        scanf_s("%d", &no);
        do
        {
            printf("변환할 진수를 입력하라.:");
            scanf_s("%d", &cd);
        } while (cd < 2 || cd > 36);
        dno = card_conv(no, cd, cno);
        printf("10진수 %d는 %d진수로는 ", no, cd);
        for (int i = dno - 1; i >= 0; i--)
        {
            printf("%c", cno[i]);
        }
        printf("\n");
        printf("다시?(1:OK/0:N)");
        scanf_s("%d", &retry);
    } while (retry);
    return 0;
}
