#ifdef _MSC_VER
#define _CRT_SECURE_NO_WARNINGS
#endif
#include 

#define N 5

int main()
{
    int i;
    int a[N];

    for(i = 0 ; i < N ; i++)
    {
        printf("a[%d] : ", i);
        scanf("%d", &a[i]);
    }
    puts("각 요소의 값");
    for (i = 0; i < N; i++)
    {
        printf("a[%d]  = %d \n", i, a[i]);
    }
    return 0;
}


#ifdef _MSC_VER
#define _CRT_SECURE_NO_WARNINGS
#endif
#include 

int main()
{
    int i;
    int a[5] = { 1,2,3,4,5 };
    int na = sizeof(a) / sizeof(a[0]); // 요소수

    printf("배열a의 요소수는 %d이다.\n", na);

    for(i = 0 ; i < na ; i++)
        printf("a[%d] :%d\n", i, a[i]);
    return 0;
}
