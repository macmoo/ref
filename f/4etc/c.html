// calloc사용
int main()
{
    int* x;
    x = calloc(1, sizeof(int));

    if (x == NULL)
        puts("영역확보에 실패했다.");
    else {
        *x = 57;
        printf("*x = %d\n", *x);
        free(x);
    }

}

int main(void)
{
    int i;
    int* a; // 배열의 선두요소의 포인터
    int na;

    printf("요소수 :");
    scanf("%d", &na);

    a = calloc(na, sizeof(int));

    if (a == NULL)
        puts("영역확보실패");
    else
    {
        printf("%d개의 정수를 입력하시오:\n", na);
        for (i = 0; i < na; i++)
        {
            printf("a[%d]", i);
            scanf("%d", &a[i]);
        }
        printf("-- 출력 --\n");
        for (int i = 0; i < na; i++)
            printf("a[%d] = %d\n", i, a[i]);
        free(a); // 요소수 na의 int형 배열을 파기
    }
    return 0;
}

int maxOf(const int a[], int n)
{
    int i;
    int max = a[0];
    for (i = 1; i < n; i++)
        if (a[i] > max) max = a[i];
    return max;
}
