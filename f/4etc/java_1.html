// 문자열 패턴 체크
private static boolean isPatternMatched(String target, String ptn) {
    if(StringUtils.isBlank(target)) {
        return true;
    }
    Pattern pattern = Pattern.compile(ptn);
    Matcher matcher = pattern.matcher(target);
    return matcher.matches();
}
// 데이트 유효성 체크
private static boolean isDateValid(String target, String ptn) {
    String afterConvert = null;
    try {
        DateTimeFormatter dtf = DateTimeFormatter.ofPattern(ptn);
        afterConvert = dtf.format(LocalDate.parse(target, dtf));
    } catch (DateTimeParseException dtp) {
        return false;
    }
    return StringUtils.equals(target, afterConvert);
}
// 문자열 마지막의 개행코드(\r\n)확인
private static boolean checkCRLF(String target) {
    if( target.charAt(target.length() - 2 ) == (char)13
        && target.charAt(target.length() - 1 ) == (char)10 ) {
        return true;
    }
    return false;
}
